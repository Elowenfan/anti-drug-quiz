  <script>
    function checkAnswers() {
      let score = 0;

      // 題目1正確答案: A, B, D
      let q1Answers = Array.from(document.querySelectorAll('input[name="q1"]:checked')).map(i => i.value);
      let correctQ1 = ["A","B","D"];
      if (q1Answers.length === correctQ1.length && q1Answers.every(val => correctQ1.includes(val))) {
        score += 33;
      }

      // 題目2正確答案: B (第二級)
      if (document.querySelector('input[name="q2"]:checked')?.value === "B") score += 33;

      // 題目3正確答案: C (0800-770-885)
      if (document.querySelector('input[name="q3"]:checked')?.value === "C") score += 34;

      let result = document.getElementById("result");
      let formSection = document.getElementById("formSection");

      if (score === 100) {
        result.textContent = "💯 恭喜！你答對了，獲得 100 分，可以參加抽獎！";
        result.style.color = "green";
        formSection.style.display = "block";
      } else {
        result.textContent = "😏 分數：" + score + " 分，還差一點，不能參加抽獎喔！";
        result.style.color = "red";
        formSection.style.display = "none";
      }
    }

    document.getElementById("lotteryForm").addEventListener("submit", function(event) {
      event.preventDefault();
      document.getElementById("submitResult").textContent = "✅ 已成功報名抽獎！祝你好運 🍀";
    });
  </script>
